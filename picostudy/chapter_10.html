Chapter 10. 외부 입력 통신 (내장 블루투스, IR 리모컨)

본 가이드북의 최종장입니다. Pico 2 W에 내장된 무선 안테나를 깨워 스마트폰과 1:1로 비밀 통신을 나누는 블루투스 기술과, 가전제품의 핵심인 적외선(IR) 리모컨 신호 디코딩 제어 기법을 마스터해 봅니다.

---

10.1 내장 블루투스(BLE) 세팅 방법과 원리

Pico 2 W는 무선 LAN 칩셋 안에 Bluetooth Low Energy (BLE) 기능을 기본 탑재하고 있습니다. 과거 HC-06 같은 외장 모듈처럼 Classic Bluetooth 통신을 쓰는 것이 아니라, 전력을 극도로 아끼는 최신 BLE 표준 프로토콜을 사용합니다.

💡 스마트폰 연동 전 필수 체크 사항

BLE는 일반 블루투스처럼 스마트폰의 기본 [설정] -> [블루투스] 메뉴에서 검색하면 잡히지 않습니다! 반드시 플레이스토어나 앱스토어에서 "Serial Bluetooth Terminal" 이나 "nRF Connect" 같은 전문 BLE 지원 통신 앱을 설치하여 앱 안에서 장치를 검색하고 연결해야 합니다.

10.2 블루투스(BLE) 연동 기본 코드

마이크로파이썬 표준 ubluetooth 모듈을 사용해 내 보드를 주변 스마트폰이 검색할 수 있도록 무선 신호(Advertising)를 뿌려주고, 가상의 시리얼 포트(Nordic UART 서비스)를 개방하는 표준 기본 베이스라인 코드입니다.

import bluetooth
import time
from micropython import const

# BLE 이벤트 상수 설정
_IRQ_CENTRAL_CONNECT = const(1)
_IRQ_CENTRAL_DISCONNECT = const(2)
_IRQ_GATTS_WRITE = const(3)

# 표준 Nordic UART 서비스(NUS) UUID 정의
_UART_UUID = bluetooth.UUID("6E400001-B5A3-F393-E0A9-E50E24DCCA9E")
_UART_TX = (bluetooth.UUID("6E400003-B5A3-F393-E0A9-E50E24DCCA9E"), bluetooth.FLAG_NOTIFY,)
_UART_RX = (bluetooth.UUID("6E400002-B5A3-F393-E0A9-E50E24DCCA9E"), bluetooth.FLAG_WRITE_NO_RESPONSE,)
_UART_SERVICE = (_UART_UUID, (_UART_TX, _UART_RX),)

class PicoBLEUART:
    def __init__(self, name="Pico2W_BLE"):
        self._ble = bluetooth.BLE()
        self._ble.active(True)
        self._ble.irq(self._irq)
        ((self._tx_handle, self._rx_handle),) = self._ble.gatts_register_services((_UART_SERVICE,))
        self._connections = set()
        self._advertise(name)
        print(f"[{name}] 블루투스 신호 송출 중... 페어링을 대기합니다.")

    def _irq(self, event, data):
        if event == _IRQ_CENTRAL_CONNECT:
            conn_handle, _, _ = data
            self._connections.add(conn_handle)
            print("\n⚡ 스마트폰 연결 성공!")
        elif event == _IRQ_CENTRAL_DISCONNECT:
            conn_handle, _, _ = data
            self._connections.remove(conn_handle)
            print("\n❌ 연결이 해제되었습니다. 다시 대기합니다.")
            self._advertise()
        elif event == _IRQ_GATTS_WRITE:
            conn_handle, value_handle = data
            if value_handle == self._rx_handle:
                msg = self._ble.gatts_read(self._rx_handle).decode('utf-8').strip()
                self.on_rx(msg)

    def _advertise(self, name="Pico2W_BLE"):
        payload = b'\x02\x01\x06' + bytes([len(name) + 1, 0x09]) + name.encode('utf-8')
        self._ble.gap_advertise(100000, adv_data=payload)

    def send(self, data):
        for conn_handle in self._connections:
            self._ble.gatts_notify(conn_handle, self._tx_handle, data + '\n')

    def on_rx(self, msg):
        # 수신된 메시지를 터미널 창에 그대로 출력 (기본 동작)
        print(f"📱 스마트폰 수신 데이터: {msg}")

# 가동 테스트
ble = PicoBLEUART()

10.3 블루투스 연동 응용 예제 코드 (무선 원격 제어)

위에서 구현한 BLE 통신 클래스 구조를 상속받아, 스마트폰 앱에서 무선으로 on 또는 off 텍스트 명령을 전달했을 때 내장 LED를 제어하고 보드가 스마트폰 화면으로 처리 완료 답장을 보내주는 실전 사물인터넷 응용 스크립트입니다.

from machine import Pin
import time

# 10.2에서 만든 클래스를 고스란히 확장하여 사용합니다.
led = Pin("LED", Pin.OUT)

class CustomIoTController(PicoBLEUART):
    def on_rx(self, msg):
        print(f" 명령 접수: {msg}")
        
        if msg == "on":
            led.value(1)
            self.send("Pico2W > LED가 켜졌습니다!")
            print("-> 하드웨어 작동: LED ON")
        elif msg == "off":
            led.value(0)
            self.send("Pico2W > LED가 꺼졌습니다!")
            print("-> 하드웨어 작동: LED OFF")
        else:
            self.send(f"unknown command: {msg}")

# 실전 스마트홈 컨트롤러 가동
iot_server = CustomIoTController("Pico2W_SmartHome")

while True:
    time.sleep(1) # 통신 이벤트 대기를 위한 루프 유지

10.4 IR 적외선 수신기 기본 세팅 방법과 원리

적외선(IR) 수신 모듈(VS1838B 기준)은 눈에 보이지 않는 940nm 파장의 적외선 빛 신호를 감지합니다. 핀 배선 시 플러스(VCC)와 마이너스(GND)가 반대로 교차하면 센서 내부 소자가 손상되므로 아래의 지도를 엄격하게 지켜 연결해야 합니다.

⚠️ 하드웨어 배선 규격 (둥근 감지 렌즈가 나를 바라보는 기준)

• 1번 핀 (맨 왼쪽, OUT): 나의 GPIO 2번 (물리 핀 4번)
• 2번 핀 (가운데, GND): 나의 GND (물리 핀 3번)
• 3번 핀 (맨 오른쪽, VCC): 나의 3V3(OUT) (물리 핀 36번)

10.5 적외선(IR) 리모컨의 동작 원리 (NEC 프로토콜)

우리가 리모컨 버튼을 누르면 내부 IC 칩이 고유 데이터(주소 및 명령)를 디지털 신호로 변환한 뒤, 38kHz 주파수로 적외선 LED를 깜빡여 전송합니다. 이를 NEC 프로토콜이라고 부르며, 신호는 시작을 알리는 '리더 펄스(약 9000us)'로 문을 열고 데이터 비트 간의 간격을 측정해 0과 1을 판별해 냅니다.

10.6 IR 수신기 연동 테스트 (펄스 마이크로초 측정)

센서가 리모컨 적외선 파형을 감지해 신호가 떨어지는(Low, 0) 순간의 총 지속 시간을 마이크로초(us) 단위로 정확히 가로채어 출력해 내는 통신 물리 계층 테스트 코드입니다.

from machine import Pin
import time

# GPIO 2번에 IR 수신 핀 배치
ir_sensor = Pin(2, Pin.IN)

print("리모컨 적외선 수신 대기중... 버튼을 눌러보세요.")

while True:
    # 수신 센서 파형 신호가 Low로 요동치기 시작하면 (신호 감지 순간)
    if ir_sensor.value() == 0:
        pulse_start = time.ticks_us()
        
        # 신호 지속 처리 감지
        while ir_sensor.value() == 0:
            pass
        pulse_end = time.ticks_us()
        
        duration = pulse_end - pulse_start
        # 펄스 신호의 길이에 따라 리모컨 고유 식별 코드가 생성됩니다.
        print(f"적외선 신호 포착! 고유 펄스 길이: {duration} us")
        
        time.sleep(0.2) # 연속 중복 입력 필터링 여유 대기
    time.sleep(0.01)

10.7 리모컨 연동 동작 구현 및 응용 (가전제품 제어 모델)

10.6의 테스트를 통해 리모컨의 특정 버튼을 눌렀을 때의 펄스 길이를 파악했다면, 그 수치 데이터 범위를 파이썬의 오차 범위 조건문(if) 설계로 가두어 특정 버튼을 누를 때마다 내장 LED를 토글(Toggle) 전환하는 완벽한 가전 원격 제어 실습입니다.

from machine import Pin
import time

ir_sensor = Pin(2, Pin.IN)
target_led = Pin("LED", Pin.OUT)

print("스마트 리모컨 제어 시스템 가동...")

while True:
    if ir_sensor.value() == 0:
        pulse_start = time.ticks_us()
        while ir_sensor.value() == 0:
            pass
        pulse_end = time.ticks_us()
        
        duration = pulse_end - pulse_start
        
        # [실전 응용] 리모컨의 특정 버튼 리더 펄스가 대략 8500~9500us 사이로 들어온다면!
        if 8500 < duration < 9500:
            print(f"인증된 리모컨 버튼 입력 확인! ({duration} us)")
            target_led.toggle() # 내장 LED를 반전시킵니다.
            print("-> 상태 제어: LED 상태가 변경되었습니다.")
            
        time.sleep(0.25)
    time.sleep(0.01)
