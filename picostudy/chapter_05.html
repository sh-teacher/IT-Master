Chapter 5. 실전 IoT: 웹 기반 원격 LED 제어

우리가 와이파이를 연결한 진짜 이유입니다. 보드가 직접 웹 서버가 되어, 같은 네트워크 안의 스마트폰이나 PC의 브라우저 명령을 받아 동작하는 사물인터넷(IoT) 시스템의 기초를 다져봅니다.

네트워크망을 이용해 무선으로 하드웨어를 통제하는 기술

5.1 단계별 웹 서버 빌드업 실습

단번에 복잡한 코드를 실행하기보다, 통신의 흐름을 이해하기 위해 쉬운 단계부터 접근해 보겠습니다. (※ Chapter 4에서 배운 와이파이 접속 코드가 기본적으로 포함됩니다.)

[1단계] 웹 브라우저 접속 감지하고 한 줄 인사하기

소켓(Socket)을 열고, 사용자가 인터넷 창에 보드의 IP 주소를 치고 들어왔을 때 발생하는 '요청 신호'를 셸 창에 출력한 뒤 간단한 영문 텍스트를 던져주는 가장 단순한 서버 구조입니다.

import network
import socket
import time

ssid = '본인의_와이파이_이름'
password = '본인의_비밀번호'

# Wi-Fi 연결 프로세스
wlan = network.WLAN(network.STA_IF)
wlan.active(True)
wlan.connect(ssid, password)
while not wlan.isconnected():
    time.sleep(0.5)
print('와이파이 연결 성공! IP 주소:', wlan.ifconfig()[0])

# 웹 서버 대기 (Socket)
addr = socket.getaddrinfo('0.0.0.0', 80)[0][-1]
s = socket.socket()

# ◀ [핵심] 이 한 줄을 bind 직전에 추가해줘! 포트 재사용을 허용하는 명령이야.
s.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1)

s.bind(addr)
s.listen(1)

print("1단계 기본 통신 서버 가동...")

while True:
    # 누군가 접속할 때까지 멈춰서 기다립니다. (Accept)
    cl, client_addr = s.accept()
    print("클라이언트 접속 감지! 주소:", client_addr)
    
    # 사용자가 요청한 데이터(HTTP Request)를 읽어옵니다.
    request = cl.recv(1024).decode('utf-8')
    print("--- 받은 요청문 ---")
    print(request)
    print("-------------------")
    
    # 접속한 브라우저 화면에 간단한 문자열을 전달하고 소켓을 닫습니다.
    cl.send('HTTP/1.0 200 OK\r\n\r\n')
    cl.send('Hello! Connection Success from Pico 2.')
    cl.close()

[2단계] 한글 글자 깨짐 해결 및 웹 화면 꾸미기

마이크로파이썬 서버에서 한글이 깨지는 이유는 웹 브라우저에게 한글 규격방식인 UTF-8로 해석하라는 명시적 헤더를 주지 않았고, 문자열을 바이트 형태로 인코딩하지 않았기 때문입니다. 이를 보완해 완벽한 한글 웹페이지를 띄우는 코드입니다.

import network
import socket
import time

ssid = '본인의_와이파이_이름'
password = '본인의_비밀번호'

wlan = network.WLAN(network.STA_IF)
wlan.active(True)
wlan.connect(ssid, password)
while not wlan.isconnected():
    time.sleep(0.5)
ip = wlan.ifconfig()[0]
print(f'2단계 서버 가동! 접속 주소: http://{ip}')

addr = socket.getaddrinfo('0.0.0.0', 80)[0][-1]
s = socket.socket()
s.bind(addr)
s.listen(1)

# 사용자에게 띄워줄 HTML 화면 구조 정의
def get_html():
    return """<!DOCTYPE html>
    <html>
    <head><meta charset="UTF-8"></head>
    <body style="text-align:center; font-family:sans-serif; padding-top:50px;">
        <h1 style="color:#2563eb;">Pico 2 무선 웹 서버 웹사이트</h1>
        <p>보드가 직접 생성한 데이터가 정상적으로 출력되고 있습니다.</p>
        <p style="font-weight:bold; color:#10b981;">한글 깨짐 현상이 완전히 해결되었습니다!</p>
    </body>
    </html>"""

while True:
    try:
        cl, addr_info = s.accept()
        request = cl.recv(1024).decode('utf-8')
        
        # [핵심] Content-Type 헤더에 charset=utf-8을 명시해 브라우저에게 안내합니다.
        cl.send('HTTP/1.1 200 OK\r\nContent-Type: text/html; charset=utf-8\r\n\r\n'.encode('utf-8'))
        
        # [핵심] HTML 스트링 문자열을 .encode('utf-8')을 통해 안전하게 변환하여 송신합니다.
        cl.send(get_html().encode('utf-8'))
        cl.close()
    except Exception as e:
        print("에러 발생:", e)
        cl.close()

[3단계] 실전 IoT: 웹 쿼리 분석을 통한 원격 LED 제어 (완성형)

이제 화면에 ON/OFF 버튼을 만들고, 버튼을 누를 때 주소창 뒤에 붙는 파라미터(/?led=on과 /?led=off) 문자열을 보드가 실시간으로 해부 및 검사하여 실제 하드웨어 핀을 물리적으로 켜고 끄는 마스터 코드입니다.

import network
import socket
import time
from machine import Pin

ssid = '본인의_와이파이_이름'
password = '본인의_비밀번호'

# 내장 LED 제어 핀 생성
led = Pin("LED", Pin.OUT)

wlan = network.WLAN(network.STA_IF)
wlan.active(True)
wlan.connect(ssid, password)
while not wlan.isconnected():
    time.sleep(0.5)
ip = wlan.ifconfig()[0]
print(f'★실전 IoT 서버 가동! 접속 주소: http://{ip}')

addr = socket.getaddrinfo('0.0.0.0', 80)[0][-1]
s = socket.socket()
s.bind(addr)
s.listen(1)

# 제어 상태를 시각적으로 보여줄 동적 UI 생성 함수
def get_iot_html(led_status):
    # LED 상태에 따라 다르게 보일 라벨 색상 지정
    status_color = "#10b981" if led_status == "켜짐" else "#ef4444"
    
    return f"""<!DOCTYPE html>
    <html>
    <head>
        <meta charset="UTF-8">
        <meta name="viewport" content="width=device-width, initial-scale=1.0">
        <title>Pico 2 IoT Control</title>
    </head>
    <body style="text-align:center; font-family:sans-serif; background-color:#f8fafc; padding-top:50px;">
        <h1 style="color:#1e293b;">Pico 2 W 원격 제어 센터</h1>
        <div style="margin:20px; font-size:20px;">현재 LED 상태: <span style="color:{status_color}; font-weight:bold;">{led_status}</span></div>
        <div style="margin-top:40px;">
            <a href="/?led=on" style="text-decoration:none;">
                <button style="font-size:24px; padding:15px 35px; background:#10b981; color:white; border:none; border-radius:10px; cursor:pointer; margin:10px; font-weight:bold;">LED 켜기 (ON)</button>
            </a>
            <a href="/?led=off" style="text-decoration:none;">
                <button style="font-size:24px; padding:15px 35px; background:#ef4444; color:white; border:none; border-radius:10px; cursor:pointer; margin:10px; font-weight:bold;">LED 끄기 (OFF)</button>
            </a>
        </div>
    </body>
    </html>"""

# 현재 핀의 상태값 저장용 변수
current_status = "꺼짐"

while True:
    try:
        cl, addr_info = s.accept()
        request = cl.recv(1024).decode('utf-8')
        
        # 브라우저가 전송한 HTTP 요청문 안에 주소 파라미터 키워드가 있는지 감별합니다.
        if '/?led=on' in request:
            led.value(1)
            current_status = "켜짐"
            print("웹 명령 접수: LED ON")
        elif '/?led=off' in request:
            led.value(0)
            current_status = "꺼짐"
            print("웹 명령 접수: LED OFF")
            
        # 응답 규격 송신 및 HTML 바이트 변환 출력
        cl.send('HTTP/1.1 200 OK\r\nContent-Type: text/html; charset=utf-8\r\n\r\n'.encode('utf-8'))
        cl.send(get_iot_html(current_status).encode('utf-8'))
        cl.close()
    except Exception as e:
        cl.close()

5.2 웹 제어 작동 확인

  1. 최종 3단계 코드를 실행하고 하단 콘솔 셸 창에 찍히는 http://192.168.x.x 형태의 주소를 확인합니다.
  2. 보드와 반드시 같은 와이파이(공유기)에 물려 있는 스마트폰이나 노트북의 인터넷 주소창에 해당 주소를 입력하고 접속합니다.
  3. 한글이 깨지지 않고 깔끔하게 열리는 대시보드 화면을 마주했다면 성공입니다. 버튼을 누르면 나의 내장 LED가 즉각적으로 연동하여 켜지고 꺼집니다.
